1function [x, rnorme, rnorml, exitflag] = mlsei(C, d, A, b, Aeq, beq, lb, ub)
  2% mlsei: Constrained linear least squares based on the lsei algoithm from
  3% the slatec fortran library.
  4%
  5% X = mlsei(C,d,A,b) attempts to solve the least-squares problem
  6% 
  7%         min  0.5*(NORM(C*x-d)).^2       subject to    A*x <= b
  8%          x
  9% 
 10% where C is m-by-n.
 11% 
 12% X = mlsei(C,d,A,b,Aeq,beq) solves the least-squares
 13% (with equality constraints) problem:
 14% 
 15%         min  0.5*(NORM(C*x-d)).^2    subject to 
 16%          x                               A*x <= b and Aeq*x = beq
 17% 
 18% X = mlsei(C,d,A,b,Aeq,beq,LB,UB) defines a set of lower and upper
 19% bounds on the design variables, X, so that the solution
 20% is in the range LB <= X <= UB. Set LB(i) = -Inf if X(i) is unbounded 
 21% below; set UB(i) = Inf if X(i) is unbounded above.
 22%
 23%
 24% mlsei is a gateway function to the fortran routine dlsei, the
 25% double-precision version of the lsei algorithm from the slatec
 26% fortran mathemetical library available here:
 27%
 28%   http://www.netlib.org/slatec/
 29%
 30% For a detailed description of the algorithm used see:
 31%
 32% Richard J. Hanson and Karen H. Haskell. 1982. Algorithm 587: Two
 33% Algorithms for the Linearly Constrained Least Squares Problem. ACM Trans.
 34% Math. Softw. 8, 3 (September 1982), 323-333. DOI=10.1145/356004.356010
 35% http://doi.acm.org/10.1145/356004.356010 
 36%
 37% The required functions have been compiled into mex functions for several
 38% platforms and are accessed through the mexlsei routine. Unfortunately
 39% this means mlsei currently only works on the following architectures:
 40% win32, win64, glx, glxa64.
 41%
 42
 43% Copyright Richard Crozier 2011
 44
 45    % get the number of variables
 46    n = size(C, 2);
 47    
 48    if nargin == 2
 49        
 50        A = [];
 51        b = [];
 52        Aeq = [];
 53        beq = [];
 54        
 55    elseif nargin == 4
 56        
 57        Aeq = [];
 58        beq = [];
 59    
 60    elseif nargin > 6
 61        
 62        % Bounds on the variables can be handled as additional 
 63        % linear constaints
 64        
 65        if length(lb) ~= n
 66            error('lb vector must be of same size as number of variables, set unbounded variables to -InF.')
 67        elseif length(ub) ~= n
 68            error('ub vector must be of same size as number of variables, set unbounded variables to InF')
 69        end
 70        
 71        % get the members of lb that are not equal to -inf (or inf)
 72        noninflb = ~isinf(lb);
 73
 74        if nnz(noninflb) > 0
 75
 76            % create an n x n matrix with diagonal line
 77            % of negative ones and zeros elswhere
 78            lbmatrix = -eye(n);
 79
 80            % add non-Inf bounds to C matrix
 81            A = [A; lbmatrix(noninflb, :)];
 82
 83            % add the constraints to the b vector
 84            b = [b; -lb(noninflb)];
 85
 86        end
 87
 88        % upper bounds
 89        noninfub = ~eq(ub,inf);
 90        
 91        if nnz(noninfub) > 0
 92            
 93            % create an n x n matrix with diagonal line
 94            % of one and zeros elswhere
 95            ubmatrix = eye(n);
 96            
 97            % add non-Inf bounds to C matrix
 98            A = [A; ubmatrix(noninfub,:)];
 99            
100            b = [b; ub(noninfub)];
101            
102        end
103
104    end
105
106    % Build the w matrix for dlsei
107    w = [Aeq, beq;
108         C,   d;
109        -A,  -b];
110    
111    mdw = size(w,1);
112    me = size(Aeq,1);
113    ma = size(C,1);
114    mg = size(A,1);
115    
116    % the options for lsei
117    prgopt = 1;
118    
119    % dlsei requires working space to be provided in
120    % ws
121    k = max(ma+mg, n);
122    wsize = 2*(me+n)+k+(mg+2)*(n+7);
123    ws = zeros(1,wsize);
124    
125    % dlsei requires working space to be provided in ip 
126    %
127    % From lsei doc:
128    %
129    % Before computation
130    %
131    %     IP(1),       The amounts of working storage actually 
132    %     IP(2)        allocated for the working arrays WS(*) and 
133    %                  IP(*), respectively.  These quantities are 
134    %                  compared with the actual amounts of storage 
135    %                  needed by DLSEI( ).  Insufficient storage allocated for either WS(*) or IP(*) is an */
136    %                  error.  This feature was included in DLSEI( ) */
137    %                  because miscalculating the storage formulas */
138    %                  for WS(*) and IP(*) might very well lead to */
139    %                  subtle and hard-to-find execution errors. */
140    %
141    %                  The length of WS(*) must be at least */
142    %
143    %                  LW = 2*(ME+N)+K+(MG+2)*(N+7) */
144    %
145    %                  where K = max(MA+MG,N) */
146    %                  This test will not be made if IP(1).LE.0. */
147    %
148    %                  The length of IP(*) must be at least */
149    %
150    %                  LIP = MG+2*N+2 */
151    %                  This test will not be made if IP(2).LE.0. */
152    %
153    % IP   The integer working array has three entries that provide rank 
154    %      and working array length information after completion.
155    %
156    %      IP(1) = rank of equality constraint matrix.              
157    %
158    %      IP(2) = rank of reduced least squares problem.              
159    %
160    %      IP(3) = the amount of storage in the working array WS(*) that
161    %              was actually used by the subprogram. The formula given
162    %              above for the length of WS(*) is a necessary
163    %              overestimate. If exactly the same problem matrices are
164    %              used in subsequent executions, the declared dimension of
165    %              WS(*) can be reduced to this output value.
166    
167    ipsize = mg+2*n+2;
168    ip = zeros(1, ipsize);
169    
170    % We pass in the size of the 
171    ip(1) = wsize;
172    ip(2) = ipsize;
173    
174    % construct the input vector W
175    w = reshape(w, 1, []);
176
177    % call dlsei via the mex gateway function mexlsei
178    [x, rnorme, rnorml, exitflag] = mexlsei(w, mdw, me, ma, mg, n, prgopt, ws, ip);
179    
180    if exitflag
181        exitflag = -exitflag;
182    end
183
184end
